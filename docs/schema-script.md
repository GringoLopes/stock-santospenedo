AS $$

  RETURN crypt(password, gen_salt('bf', 8));

-- Fun√ß√£o para atualizar senha de um usu√°rio

      updated_at = now()

AS $$

  -- S√≥ hasheia se a senha foi alterada
  IF TG_OP = 'INSERT' OR new.password != old.password THEN

  RETURN new;
